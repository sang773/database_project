🔬 SQL Queries Behind This Page

Key Database Queries

1 · Category Aggregates (GROUP BY + LEFT JOIN)
SELECT c.CategoryName,
       COUNT(l.ListingID)      AS active_count,
       ROUND(AVG(l.Price), 2)  AS avg_price,
       ROUND(MIN(l.Price), 2)  AS min_price,
       ROUND(MAX(l.Price), 2)  AS max_price
FROM Category c
LEFT JOIN Listing l ON c.CategoryID = l.CategoryID AND l.Status = 'active'
GROUP BY c.CategoryID, c.CategoryName
ORDER BY avg_price DESC;
2 · Hot Deals — Subquery JOIN + CASE
SELECT l.Title, l.Price,
       ROUND((1 - l.Price / ca.AvgPrice) * 100, 0) AS PctBelow
FROM Listing l
JOIN (
    SELECT CategoryID, AVG(Price) AS AvgPrice
    FROM Listing WHERE Status = 'active'
    GROUP BY CategoryID
    HAVING AVG(Price) > 0
) ca ON ca.CategoryID = l.CategoryID
WHERE l.Status = 'active'
  AND l.Price < ca.AvgPrice * 0.80
ORDER BY PctBelow DESC;
3 · Recent Sales — 5-Table JOIN
SELECT t.Date, t.FinalPrice, l.Title,
       buyer.Name AS BuyerName, seller.Name AS SellerName
FROM Transaction t
JOIN Listing  l      ON t.ListingID  = l.ListingID
JOIN Category c      ON l.CategoryID = c.CategoryID
JOIN Student  buyer  ON t.BuyerID    = buyer.StudentID
JOIN Student  seller ON t.SellerID   = seller.StudentID
ORDER BY t.Date DESC LIMIT 8;
4 · Top Sellers — GROUP BY + HAVING
SELECT s.Name, ROUND(s.Rating,1) AS Rating,
       COUNT(DISTINCT r.ReviewID) AS review_count
FROM Student s
LEFT JOIN Review r ON s.StudentID = r.RevieweeID
GROUP BY s.StudentID, s.Name, s.Rating
HAVING review_count > 0
ORDER BY s.Rating DESC;
5 · Deal Score per Listing — CASE + Correlated Subquery
SELECT l.Title, l.Price,
       CASE
         WHEN l.Price < ca.AvgPrice * 0.80 THEN 'Hot'
         WHEN l.Price > ca.AvgPrice * 1.20 THEN 'High'
         ELSE 'Fair'
       END AS DealScore
FROM Listing l
LEFT JOIN (
    SELECT CategoryID, AVG(Price) AS AvgPrice
    FROM Listing WHERE Status = 'active'
    GROUP BY CategoryID
) ca ON ca.CategoryID = l.CategoryID
WHERE l.Status = 'active';
6 · Most-Saved Listings — Watchlist × Listing JOIN + GROUP BY
SELECT l.Title, l.Price, c.CategoryName,
       COUNT(w.WatchlistID) AS save_count
FROM Listing l
JOIN Category  c ON l.CategoryID = c.CategoryID
JOIN Student   s ON l.StudentID  = s.StudentID
JOIN Watchlist w ON l.ListingID  = w.ListingID
WHERE l.Status = 'active'
GROUP BY l.ListingID, l.Title, l.Price, c.CategoryName, s.Name
ORDER BY save_count DESC LIMIT 5;
7 · Personalized Recommendations — Correlated Subqueries + 5-Table Query
SELECT l.Title, l.Price, c.CategoryName
FROM Listing l
JOIN Student s ON l.StudentID = s.StudentID
JOIN Category c ON l.CategoryID = c.CategoryID
WHERE l.Status = 'active'
  AND l.StudentID != :current_user
  AND l.ListingID NOT IN (
      SELECT ListingID FROM Watchlist
      WHERE StudentID = :current_user   -- exclude already-saved
  )
  AND l.CategoryID IN (
      SELECT DISTINCT l2.CategoryID    -- categories user has saved
      FROM Watchlist w
      JOIN Listing l2 ON w.ListingID = l2.ListingID
      WHERE w.StudentID = :current_user
  )
ORDER BY DealScore ASC, DatePosted DESC LIMIT 4;
